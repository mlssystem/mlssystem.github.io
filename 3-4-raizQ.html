Exercício 2

algoritmo "RaizQ - Interativo"

var

N1, R: real
N, RESP, NOME: caractere

inicio

repita Escreval("Qual seu NOME?")
leia (NOME)
limpatela
Escreval (NOME,", digite um número:")
Escreva("Nº: ")
leia(N1)
limpatela
Escreval("Você digitou o número ",N1)
Escreval("")
R <- RaizQ(N1)
Escreval (NOME,", a raiz quadrada de", N1, " é: ",R)
Escreval ("Deseja continuar ",NOME," [S/N]?")
Escreva("Resposta: ")
leia(RESP)
ate(RESP="N")
limpatela
Escreval("")
Escreval("--------------------FIM-------------------")
Escreval(" Volte sempre ", NOME,"." )
